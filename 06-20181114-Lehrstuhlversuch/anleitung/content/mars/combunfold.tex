\subsection{CombUnfold}%
\label{sub:combunfold}
\paragraph{Theorie}%
\label{par:theorie}

\begin{equation}
	g(y) = \int_\text{c}^\text{d} M(x,y) f(x) \text{dx} + b(y)
\end{equation}

\begin{equation}
	g_i = \sum_j M_{ij} f_j + b_i
\end{equation}

\begin{equation}
	g_i = \int_{y_{i-1}}^{y_i} g(y) \text{dy} \quad \text{und} \quad
	b_i = \int_{y_{i-1}}^{y_i} b(y) \text{dy}
\end{equation}
